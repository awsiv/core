Cfengine 3 example unit_getenv.html


body common control
{
bundlesequence => { "test" };
}

bundle agent test
{
vars:

   "myvar" string => getenv("PATH","20");

classes:

  "isdefined" not => strcmp("$(myvar)","");

reports:

  isdefined::

   "The path is $(myvar)";

  !isdefined::

   "The named variable PATH does not exist";

}
}
